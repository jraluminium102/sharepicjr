3. สูตรโค้ด — helper + แก้ calcUnit cabinet

3.1 helper ใหม่ — ค่าสีอลูทึบ (rn90) + ค่าสีกระจก (no-min)

// ค่าสีอลูทึบ — map สี (ci 0-12) → 1 ใน 4 ระดับ rn90 (ค่า default ตามชื่อสี · ⚠ เช็คซ้ำ)
function rn90FinRate(ci){
  if(ci===2||ci===3) return 400;            // ซาฮารา เทา/ดำ
  if(ci>=5&&ci<=7) return 1500;             // ลายไม้สต๊อก
  if(ci===8||ci===9||ci===11) return 2200;  // ลายไม้อบพิเศษ (Fuji/ระบุรหัส)
  if(ci===4||ci===10||ci===12) return 1600; // Aztec / อบพิเศษ / สีชุบ → อบสีพิเศษ
  return 0;                                  // idx 0,1 ขาว/ดำ = ไม่คิด
}
// ค่าสีกระจก — เหมือน colorPrice แต่ตัด min (มติพี่นัท: กันตู้เล็กโดน min พุ่ง)
function colorRateNoMin(ci,a,series){
  const c=COLORS[ci]; if(!c)return 0;
  let rates=c.rates;
  if(c.altFor && series && c.altFor.indexOf(series)>=0){ rates=c.altRates; }
  if(!rates) return 0;          // ขาว/ดำ = 0 · ⚠ สีชุบ idx12 rates=null ก็คืน 0 → ดู TODO
  return a*rateOf(a,rates);     // ตัด Math.max(min,...)
}

3.2 แก้ calcUnit cabinet (เพิ่ม term ค่าสีโครง)

ในบล็อก if(p.cabinet){...} หลังคำนวณ wall/fl/shc เดิม เพิ่มค่าสีโครงก่อน _sell:

// ── ค่าสีโครงตู้ (ใหม่) ── ci = สีหลัก (.i-color) · series = p.series||''
const wmat=(optSel&&optSel.cabwallmat)||'alu';
const aWall = 2*D*H + (_bw ? W*H : 0);            // พื้นที่ผนัง (ข้าง 2 + หลังถ้าติ๊ก)
const aFloor = W*D;                               // พื้น (อลูทึบเสมอ)
const nShelfAlu = (sg==='glass') ? 0 : nsh;       // ชั้นกระจกไม่คิด
const aShelfAlu = W*D*nShelfAlu;
// ผนัง: ตามวัสดุ · พื้น+ชั้นอลู: rn90 เสมอ
const cWall  = (wmat==='glass') ? colorRateNoMin(ci,aWall,p.series||'') : aWall*rn90FinRate(ci);
const cFloor = aFloor*rn90FinRate(ci);
const cShelf = aShelfAlu*rn90FinRate(ci);
const colorStruct = Math.round(cWall+cFloor+cShelf);
// เดิม: const _sell=roundUp(front+wall+fl+shc);
const _sell=roundUp(front+wall+fl+shc+colorStruct);
// + ต่อท้าย msgs: ถ้า colorStruct>0 ดันบรรทัด 'ค่าสีโครง ... = '+fmt(colorStruct)
หมายเหตุ: ci = พารามิเตอร์สีหลักที่ calcUnit รับอยู่แล้ว (signature บ.1025) · _bw, sg, nsh, D, W, H มีในบล็อก cabinet อยู่แล้ว

4. ตัวอย่างคำนวณ (นักบัญชียืนยัน)

เคส A — ตู้ผนังอลูทึบ 2 บาน 1.2×2.4 ลึก 0.6 · 5 ชั้นอลู · ติ๊กผนังหลัง · สีอบพิเศษ (idx10)

  • ผนัง = 2×0.6×2.4 + 1.2×2.4 = 5.76 ตร.ม. → rn90 อบพิเศษ 1,600 → 9,216
  • พื้น 0.72 + ชั้นอลู 0.72×5=3.6 = 4.32 ตร.ม. → ×1,600 → 6,912
  • ค่าสีโครงรวม ≈ +16,128 → before 62,000 → after ~78,000

6. Definition of Done

  • ตู้สีขาว/ดำ (idx 0,1): ค่าสีโครง = 0 (ราคาเท่าเดิม · ห้ามเพี้ยน)
  • ตู้สีพิเศษ: ค่าสีโครงเพิ่มตามตัวอย่าง (เคส A ~+16k / A′ ~+19.5k)
  • ตู้เล็กผนังกระจก (เช่น 1.6 ตร.ม.): ไม่โดน min พุ่งเป็นหมื่น (เพราะ colorRateNoMin ตัด min)
  • สลับชิปวัสดุผนัง อลู↔กระจก → ราคาเปลี่ยน · ชั้นกระจกไม่บวกค่าสี

  • regression: สินค้าอื่นที่ใช้ colorPrice (บาน) ราคาไม่เปลี่ยน (helper เดิมยังมี min ครบ — colorRateNoMin เป็นฟังก์ชันใหม่แยก)
