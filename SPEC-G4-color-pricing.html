3. สูตรโค้ด

3.1 helper ใหม่ — ค่าสีอลูทึบ rn90 (no-min)

// map สีหลัก ci (0-12) → 1 ใน 4 ระดับ rn90 · ค่า default ตามชื่อสี ⚠ เช็คซ้ำ
function rn90FinRate(ci){
  if(ci===2||ci===3) return 400;            // ซาฮารา เทา/ดำ
  if(ci>=5&&ci<=7) return 1500;             // ลายไม้สต๊อก
  if(ci===8||ci===9||ci===11) return 2200;  // ลายไม้อบพิเศษ
  if(ci===4||ci===10||ci===12) return 1600; // Aztec / อบพิเศษ / สีชุบ → อบสีพิเศษ
  return 0;                                  // idx 0,1 ขาว/ดำ = ไม่คิด
}

กระจกใช้ colorPrice เดิม (ไม่ต้องทำ helper) · ไม่ต้องมี colorRateNoMin (มติเปลี่ยนเป็น "เหมือนงานบาน" = มี min)

3.2 แก้ calcUnit cabinet — เพิ่มค่าสีโครง + เนื้องานผนังตามวัสดุ

// ในบล็อก if(p.cabinet){...} · ci=สีหลัก (param) · _bw,sg,nsh,D,W,H มีอยู่แล้ว
const wmat=(optSel&&optSel.cabwallmat)||'alu';
const aWall = 2*D*H + (_bw ? W*H : 0);
const aFloor = W*D;
const nShelfAlu = (sg==='glass') ? 0 : nsh;       // ชั้นกระจกไม่คิดสี
const aShelfAlu = aFloor*nShelfAlu;

// ── เนื้องานผนัง: อลู 4,000 (เดิม) | กระจก = ข้อ2 ×0.7 ──
// เดิม: let wall=2*D*H*4000; if(_bw)wall+=W*H*4000;  → แทนด้วย:
let wall = (wmat==='glass')
  ? Math.max(5000, aWall*rateOf(aWall, RATES.FIX)) * 0.7   // ⚠ ×0.7 หลัง max (floor 3,500)
  : aWall*4000;

// ── ค่าสีโครง (ใหม่) ──
const cWall  = (wmat==='glass') ? colorPrice(ci,aWall,p.series||'') : aWall*rn90FinRate(ci);
const cFloor = aFloor*rn90FinRate(ci);
const cShelf = aShelfAlu*rn90FinRate(ci);
const colorStruct = Math.round(cWall+cFloor+cShelf);

const _sell=roundUp(front+wall+fl+shc+colorStruct);
// + ดัน msgs: 'ผนัง'+(wmat==='glass'?'กระจก':'อลู')+' · ค่าสีโครง '+fmt(colorStruct)
เตือนลำดับ: เขียน max(5000, area×rate) × 0.7 เป๊ะ (×0.7 หลัง max) — ห้ามเขียน max(5000, area×rate×0.7) (ผลต่างกันเมื่อผนังเล็ก)
